function proxy(url) { return `${workerUrl}/?url=${encodeURIComponent(url)}`; }

// ─── TIMING ──────────────────────────────────────────────────────────────────
function isRaceWeekend() {
  const d = new Date(), day = d.getDay(), h = d.getHours();
  return (day===5&&h>=16)||day===6||(day===0&&h<12);
}
function getIntervalMs() { return isRaceWeekend() ? 60*60*1000 : 4*60*60*1000; }

function scheduleChecks() {
  if (checkInterval) clearInterval(checkInterval);
  const ms = getIntervalMs();
  nextCheckTime = new Date(Date.now() + ms);
  // Reset window counts at each new check cycle
  windowPostCount = 0;
  queuedIds = [];
  stories.forEach(s => { if(s.status==='queued') s.status='new'; });
  saveWindowState();
  checkInterval = setInterval(() => { fetchAll(); scheduleChecks(); }, ms);
  updateQueueBar();
}

function updateNextCheck() {
  if (!nextCheckTime) return;
  const mins = Math.max(0, Math.round((nextCheckTime - Date.now()) / 60000));
  const el = document.getElementById('next-check');
  if (el) el.textContent = `Next check in ${mins}m`;
}

// ─── DEDUP ────────────────────────────────────────────────────────────────────
function titleKeywords(title) {
  const stop = new Set(['the','a','an','in','at','of','to','for','and','with','on','is','was','after','from','by','as','race','racing','grand','prix','gp']);
  return title.toLowerCase().replace(/[^a-z0-9\s]/g,'').split(/\s+/).filter(w => w.length>3 && !stop.has(w));
}
function titlesMatch(a, b) {
  const kwA = titleKeywords(a), kwB = new Set(titleKeywords(b));
  if (!kwA.length || !kwB.size) return false;
  const matches = kwA.filter(w => kwB.has(w)).length;
  return matches / Math.max(kwA.length, kwB.size) >= 0.4;
}
function alreadyOnSite(story) {
  for (const t of aaKeywords) { if (titlesMatch(story.title, t)) return true; }
  return false;
}
async function fetchAASite() {
  try {
    const res = await fetch(proxy(AA_FEED), {signal: AbortSignal.timeout(8000)});
    if (!res.ok) return 0;
    const text = await res.text();
    const xml = new DOMParser().parseFromString(text, 'text/xml');
    if (xml.querySelector('parsererror')) return 0;
    aaKeywords = new Set();
    xml.querySelectorAll('item title').forEach(el => { if(el.textContent) aaKeywords.add(el.textContent.trim()); });
    return aaKeywords.size;
  } catch(e) { return 0; }
}

// ─── STORY DETECTION ──────────────────────────────────────────────────────────
function detectType(title, desc) {
  const t = (title+' '+desc).toLowerCase();
  if (/\bwins\b|win the|race result|race report|victory|takes win|podium finish|wins at/.test(t)) return 'Race';
  if (/qualif|pole position|pole time|starts from|grid position|front row/.test(t)) return 'Qualifying';
  if (/\bfp1\b|\bfp2\b|\bfp3\b|free practice|practice session/.test(t)) return 'Practice';
  return 'News';
}
function detectDrivers(text) {
  const t = text.toLowerCase();
  return WATCHLIST.filter(d => t.includes(d.toLowerCase()) || t.includes(d.split(' ').pop().toLowerCase()));
}
function isWatchedDriver(name) {
  const n = name.toLowerCase();
  return WATCHLIST.some(d => n.includes(d.toLowerCase()) || n.includes(d.split(' ').pop().toLowerCase()));
}
function timeAgo(dateStr) {
  if (!dateStr) return '';
  const diff = Date.now() - new Date(dateStr).getTime();
  const mins = Math.floor(diff/60000);
  if (mins < 60) return `${mins}m ago`;
  const hrs = Math.floor(mins/60);
  if (hrs < 24) return `${hrs}h ago`;
  return `${Math.floor(hrs/24)}d ago`;
}

// ─── FETCH ALL FEEDS ──────────────────────────────────────────────────────────
async function fetchAll() {
  setStatus('loading', 'Checking Auto Action site...');
  document.getElementById('refresh-btn').disabled = true;

  const aaCount = await fetchAASite();
  setStatus('loading', 'Fetching news feeds...');

  const seen = new Set(stories.map(s => s.link));
  let newStories = [], ok = 0, fail = 0, deduped = 0;
  const log = [];

  for (const feed of FEEDS) {
    try {
      const res = await fetch(proxy(feed.url), {signal: AbortSignal.timeout(10000)});
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      const text = await res.text();
      const xml = new DOMParser().parseFromString(text, 'text/xml');
      if (xml.querySelector('parsererror')) throw new Error('XML parse error');
      let count = 0, skip = 0;
      xml.querySelectorAll('item').forEach(item => {
        const title = item.querySelector('title')?.textContent?.trim() || '';
        const link = item.querySelector('link')?.textContent?.trim() || '';
        const desc = item.querySelector('description')?.textContent?.replace(/<[^>]+>/g,'').trim().substring(0,400) || '';
        const pubDate = item.querySelector('pubDate')?.textContent || '';
        if (!title || !link || seen.has(link)) return;
        if (alreadyOnSite({title})) { deduped++; skip++; seen.add(link); return; }
        seen.add(link); count++;
        const type = detectType(title, desc);
        const drivers = detectDrivers(title+' '+desc);
        newStories.push({title, link, desc, pubDate, series:feed.series, type, drivers, priority:drivers.length>0, id:Math.random().toString(36).slice(2), status:'new'});
      });
      log.push({ok:true, msg:`[${count} new, ${skip} on AA] ${feed.url}`}); ok++;
    } catch(e) {
      log.push({ok:false, msg:`[${e.message}] ${feed.url}`}); fail++;
    }
  }

  newStories.sort((a,b) => (b.priority-a.priority)||(new Date(b.pubDate)-new Date(a.pubDate)));
  stories = [...newStories, ...stories].slice(0, 200);

  if (!nextCheckTime) scheduleChecks();
  else nextCheckTime = new Date(Date.now() + getIntervalMs());

  setStatus('live', `${newStories.length} new stories · ${ok} feeds OK${fail ? ` · ${fail} failed` : ''}`);
  document.getElementById('refresh-btn').disabled = false;

  // Dedup bar
  const db = document.getElementById('dedup-bar');
  if (aaCount > 0 || deduped > 0) {
    db.style.display = 'block';
    db.innerHTML = `Checked <strong>${aaCount}</strong> existing Auto Action stories &mdash; <strong>${deduped}</strong> ${deduped===1?'story':'stories'} excluded as already covered`;
  }

  // Feed log
  const logEl = document.getElementById('feed-log');
  if (fail > 0) {
    logEl.style.display = 'block';
    logEl.innerHTML = log.map(l => `<span class="${l.ok?'ok':'fail'}">${l.msg}</span>`).join('<br>');
  } else { logEl.style.display = 'none'; }

  updateCounts(); renderStories(); updateQueueBar();
}

// ─── RENDER ───────────────────────────────────────────────────────────────────
function seriesTagClass(s) {
  return {NASCAR:'tag-nascar',IndyCar:'tag-indycar',MotoGP:'tag-motogp',WRC:'tag-wrc',WEC:'tag-wec',News:'tag-news'}[s]||'tag-news';
}

function renderStories() {
  let filtered = stories.filter(s => {
    if (currentSeries !== 'all' && s.series !== currentSeries) return false;
    if (currentType !== 'all' && s.type !== currentType) return false;
    if (currentStatus === 'new' && s.status !== 'new') return false;
    if (currentStatus === 'queued' && s.status !== 'queued') return false;
    if (currentStatus === 'posted' && s.status !== 'posted') return false;
    return true;
  });

  const list = document.getElementById('stories-list');
  if (!filtered.length) {
    list.innerHTML = `<div class="empty-state"><h3>No stories found</h3><p>Hit "Check now" to fetch the latest, or adjust your filters.</p></div>`;
    return;
  }

  list.innerHTML = filtered.map(s => {
    const isQueued = s.status==='queued', isPosted = s.status==='posted';
    const queueFull = queuedIds.length >= MAX_POSTS;
    return `<div class="story-card${s.priority?' priority':''}${isPosted?' posted-card':''}" id="card-${s.id}">
      <div class="card-top">
        <div class="tags-col">
          <span class="series-tag ${seriesTagClass(s.series)}">${s.series}</span>
          <span class="type-pill">${s.type}</span>
        </div>
        <div class="card-body">
          <div class="story-title">${s.title}</div>
          <div class="story-meta">${timeAgo(s.pubDate)}</div>
          ${s.drivers.length ? `<div class="driver-tags">${s.drivers.map(d=>`<span class="driver-chip">${d}</span>`).join('')}</div>` : ''}
          <div class="card-actions">
            <button class="btn-xs" onclick="rewrite('${s.id}')">Rewrite for AA</button>
            ${!isPosted&&!isQueued ? `<button class="btn-blue" onclick="addToQueue('${s.id}')" ${queueFull?'disabled title="Queue full"':''}>+ Queue</button>` : ''}
            ${isQueued ? `<span class="queued-badge">In queue</span><button class="btn-xs" onclick="removeFromQueue('${s.id}')" style="opacity:0.6;">Remove</button>` : ''}
            ${isPosted ? `<span class="posted-badge">Posted</span>` : ''}
            <a href="${s.link}" target="_blank" class="source-link">Source &rarr;</a>
          </div>
          <div id="rw-${s.id}"></div>
        </div>
      </div>
    </div>`;
  }).join('');
}

function updateCounts() {
  const all = stories.length;
  document.getElementById('cnt-all').textContent = all;
  ['NASCAR','IndyCar','MotoGP','WRC','WEC','News'].forEach(s => {
    const el = document.getElementById('cnt-'+s);
    if (el) el.textContent = stories.filter(x=>x.series===s).length;
  });
  ['Race','Qualifying','Practice','News'].forEach(t => {
    const el = document.getElementById('cnt-type-'+t);
    if (el) el.textContent = stories.filter(x=>x.type===t).length;
  });
  document.getElementById('cnt-type-all').textContent = all;
}

function filterSeries(s) {
  currentSeries = s;
  document.querySelectorAll('[data-series]').forEach(b => b.classList.toggle('active', b.dataset.series===s));
  renderStories();
}
function filterType(t) {
  currentType = t;
  document.querySelectorAll('[data-type]').forEach(b => b.classList.toggle('active', b.dataset.type===t));
  renderStories();
}
function filterStatus(v) { currentStatus = v; renderStories(); }

function addToQueue(id) {
  if (queuedIds.length >= MAX_POSTS) return;
  const s = stories.find(x=>x.id===id); if (!s) return;
  s.status='queued'; queuedIds.push(id); updateQueueBar(); renderStories();
}
function removeFromQueue(id) {
  const s = stories.find(x=>x.id===id); if(s) s.status='new';
  queuedIds = queuedIds.filter(x=>x!==id); updateQueueBar(); renderStories();
}
async function postQueue() {
  if (!wpSettings.url||!wpSettings.user||!wpSettings.pass) { openSettings(); return; }
  for (const id of [...queuedIds]) {
    const s = stories.find(x=>x.id===id);
    if (!s||!s._rewrite) { alert(`"${s?.title||id}" needs to be rewritten first.`); continue; }
    await postSingle(id);
  }
}

// ─── REWRITE ──────────────────────────────────────────────────────────────────
function renderResultsTable(top5, champ) {
  if (!top5||!top5.length) return '';
  const rows = top5.slice(0,5).map((r,i) => {
    const w = isWatchedDriver(r.driver||'');
    return `<tr class="${w?'watched':''}"><td>${i+1}</td><td>${r.driver||''}</td><td style="color:var(--text3)">${r.team||''}</td><td style="color:var(--text3);text-align:right">${r.gap||''}</td></tr>`;
  }).join('');
  let champBlock = '';
  if (champ&&champ.length) {
    const crows = champ.slice(0,5).map((c,i) => {
      const w = isWatchedDriver(c.driver||'');
      return `<tr class="${w?'watched':''}"><td>${i+1}</td><td>${c.driver||''}</td><td style="color:var(--text3);text-align:right">${c.points||''} pts</td></tr>`;
    }).join('');
    champBlock = `<div><div class="rt-label">Championship</div><table class="rt"><thead><tr><th></th><th>Driver</th><th style="text-align:right">Points</th></tr></thead><tbody>${crows}</tbody></table></div>`;
  }
  return `<div class="results-block"><div class="results-grid">
    <div><div class="rt-label">Race result</div><table class="rt"><thead><tr><th></th><th>Driver</th><th>Team</th><th style="text-align:right">Gap</th></tr></thead><tbody>${rows}</tbody></table></div>
    ${champBlock}
  </div><div style="font-size:10px;color:var(--text3);margin-top:6px;">Blue = watched driver</div></div>`;
}

async function rewrite(id) {
  const story = stories.find(s=>s.id===id); if (!story) return;
  const rw = document.getElementById(`rw-${id}`);
  rw.innerHTML = `<div style="margin-top:10px;font-size:12px;color:#5b9fd6;display:flex;align-items:center;gap:6px;"><span class="spin"></span> Rewriting in AA style...</div>`;

  const isRace = story.type==='Race';
  const driverNote = story.drivers.length ? `Key Australian/NZ drivers mentioned: ${story.drivers.join(', ')}. Lead with the Australian angle where relevant.` : '';
  const wordCount = isRace?'600-700':(story.type==='Qualifying'||story.type==='Practice')?'350-450':'400-500';
  const resultsInstruction = isRace ? `\nAfter the story body output:\nRESULTS_JSON:\n{"top5":[{"pos":1,"driver":"Full Name","team":"Team","gap":"winner"},{"pos":2,"driver":"Full Name","team":"Team","gap":"+X.XXXs"},{"pos":3,"driver":"Full Name","team":"Team","gap":"+X.XXXs"},{"pos":4,"driver":"Full Name","team":"Team","gap":"+X.XXXs"},{"pos":5,"driver":"Full Name","team":"Team","gap":"+X.XXXs"}],"championship":[{"pos":1,"driver":"Full Name","points":000},{"pos":2,"driver":"Full Name","points":000},{"pos":3,"driver":"Full Name","points":000},{"pos":4,"driver":"Full Name","points":000},{"pos":5,"driver":"Full Name","points":000}]}\nMark uncertain values with ? suffix.` : '';

  const prompt = `You are a sub-editor for Auto Action, the Australian motorsport magazine. Fully rewrite this story in Auto Action house style.

Rules:
- Complete rewrite, fresh angle and structure. Not a paraphrase.
- Australian English throughout.
- Story type: ${story.type}. Match tone accordingly.
- Past tense for news and race reports.
- No direct quotes at all. Facts and outcomes only.
- Numbers one through nine spelled out; 10 and above as numerals.
- People named in full on first mention, then surname only.
- Safety Car always capitalised. Race, lap, round lowercase unless attached to a number (e.g. Lap 3).
- Championship and series lowercase unless part of an official name.
- No all-caps except official acronyms (NASCAR, WRC, WEC, MotoGP, IndyCar).
- No ampersands except in official names. No en dashes or em dashes.
- Short paragraphs, one idea each.
- Word target: ${wordCount} words.
${driverNote}

Output using these exact labels:
HEADLINE: [max 10 words]
SEO TITLE: [under 60 characters]
SLUG: [url-slug]
TAGS: [5 tags comma separated]
META: [under 155 characters]

STORY:
[body copy]
${resultsInstruction}

Source title: ${story.title}
Series: ${story.series}
Content: ${story.desc}`;

  try {
    const res = await fetch(`${workerUrl}/anthropic`, {
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body: JSON.stringify({model:'claude-sonnet-4-6', max_tokens:1500, messages:[{role:'user',content:prompt}]})
    });
    const data = await res.json();
    const text = data.content?.[0]?.text || '';
    const g = label => { const l = text.split('\n').find(x=>x.startsWith(label+':')); return l ? l.replace(label+':','').trim() : ''; };
    const headline=g('HEADLINE'), seoTitle=g('SEO TITLE'), slug=g('SLUG'), tags=g('TAGS'), meta=g('META');
    const si=text.indexOf('STORY:'), ri=text.indexOf('RESULTS_JSON:');
    const body = si>=0 ? text.slice(si+6, ri>0?ri:undefined).trim() : text;

    let resultsHTML = '';
    if (isRace && ri>=0) {
      try {
        const parsed = JSON.parse(text.slice(ri+13).trim().replace(/```json|```/g,'').trim());
        resultsHTML = renderResultsTable(
          (parsed.top5||[]).map(r=>({driver:r.driver,team:r.team,gap:r.gap})),
          (parsed.championship||[]).map(c=>({driver:c.driver,points:c.points}))
        );
      } catch(e){}
    }

    story._rewrite = {text, headline, seoTitle, slug, tags, meta, body, resultsHTML};
    const wpReady = wpSettings.url && wpSettings.user && wpSettings.pass;
    const limitHit = windowPostCount >= MAX_POSTS;

    rw.innerHTML = `<div class="rewrite-output">
      <div class="output-grid">
        <div class="output-field"><label>Headline</label><div class="val headline">${headline}</div></div>
        <div class="output-field"><label>Slug</label><div class="val"><code>${slug}</code></div></div>
        <div class="output-field"><label>Tags</label><div class="val">${tags}</div></div>
        <div class="output-field"><label>Meta</label><div class="val" style="font-size:11px;color:var(--text3)">${meta}</div></div>
      </div>
      <div class="story-body-output" id="body-${id}">${body}</div>
      ${resultsHTML}
      <div class="post-row">
        <button class="btn-xs" onclick="copyBody('${id}')">Copy story</button>
        <span class="copy-confirm" id="copied-${id}">Copied!</span>
        ${wpReady&&!limitHit&&story.status!=='posted' ? `<button class="btn btn-primary" style="font-size:11px;padding:4px 14px;" onclick="postSingle('${id}')">Post to WordPress</button>` : ''}
        ${limitHit&&story.status!=='posted' ? `<span class="limit-note">4-post limit reached for this window</span>` : ''}
        <span id="wpres-${id}"></span>
      </div>
    </div>`;
  } catch(e) {
    rw.innerHTML = `<div style="font-size:12px;color:var(--accent);margin-top:8px;">Rewrite failed: ${e.message}</div>`;
  }
}

function copyBody(id) {
  const el = document.getElementById(`body-${id}`); if(!el) return;
  navigator.clipboard.writeText(el.textContent).then(() => {
    const c = document.getElementById(`copied-${id}`);
    if(c){c.classList.add('show'); setTimeout(()=>c.classList.remove('show'),2000);}
  });
}

// ─── WORDPRESS ────────────────────────────────────────────────────────────────
async function postSingle(id) {
  if (windowPostCount >= MAX_POSTS) { alert('4-post limit reached for this window.'); return; }
  const story = stories.find(s=>s.id===id); if(!story||!story._rewrite) return;
  const el = document.getElementById(`wpres-${id}`);
  if(el) el.innerHTML = `<span style="font-size:11px;color:#5b9fd6;display:inline-flex;align-items:center;gap:4px;"><span class="spin"></span> Posting...</span>`;

  const {headline, slug, meta, body, resultsHTML} = story._rewrite;
  const fullContent = body + (resultsHTML ? '\n\n' + resultsHTML : '');
  const token = btoa(`${wpSettings.user}:${wpSettings.pass}`);
  try {
    const res = await fetch(`${wpSettings.url}/wp-json/wp/v2/posts`, {
      method:'POST',
      headers:{'Content-Type':'application/json','Authorization':`Basic ${token}`},
      body: JSON.stringify({title:headline, content:fullContent, slug, status:wpSettings.status||'draft', excerpt:meta})
    });
    if (res.ok) {
      const post = await res.json();
      windowPostCount++; story.status='posted'; queuedIds=queuedIds.filter(x=>x!==id);
      aaKeywords.add(headline);
      saveWindowState();
      if(el) el.innerHTML = `<span style="font-size:11px;color:var(--green);">Posted &mdash; <a href="${post.link}" target="_blank" style="color:inherit;">view post</a></span>`;
      updateQueueBar(); renderStories();
    } else {
      const err = await res.json();
      if(el) el.innerHTML = `<span style="font-size:11px;color:var(--accent);">WP error: ${err.message||res.status}</span>`;
    }
  } catch(e) {
    if(el) el.innerHTML = `<span style="font-size:11px;color:var(--accent);">Could not reach WordPress.</span>`;
  }
}

  mode.textContent = rw ? 'Race weekend — hourly' : 'Quiet period — 4 hourly';
  mode.className = 'queue-mode' + (rw?' race':'');
  const pqb = document.getElementById('post-queue-btn');
  pqb.style.display = queuedIds.length>0 && wpSettings.url ? 'block' : 'none';
}

  stories.forEach(s=>{if(s.status==='queued')s.status='new';});
  saveWindowState(); updateQueueBar(); renderStories();
}

function saveWindowState() {
